// Touch XPT2046 — LVGL v9 indev
void my_touchpad_read(
  lv_indev_t *indev,
  lv_indev_data_t *data)
{
  data->state = my_touch.Pressed()
    ? LV_INDEV_STATE_PR
    : LV_INDEV_STATE_REL;
  data->point.x = my_touch.X();
  data->point.y = my_touch.Y();
}
